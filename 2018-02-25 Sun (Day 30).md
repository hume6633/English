## 汇总单词的python代码
```python
import re
import glob
import os
from pprint import pprint
filenames = glob.glob('2018*.md')
def read(filename):
    with open(filename) as f:
        content = f.read()
    return content
d = {}
for filename in filenames:
    words = read(filename)
    words_list = re.findall(r'\n\*\*([\w\s]+)\*\*', words)
    words_list = [i for i in words_list if not i.endswith(('Origin', 'Etymology','Wikipedia','光系列','众神','进入正题','WB', 'PMBOK'))]
    d.update({filename:words_list})
pprint(d)
l = []
for value in d.values():
    l.extend(value)
l = list(enumerate(list(set(l))))
pprint(l)
```
